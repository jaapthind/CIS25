#include <iostream>
using namespace std;

int main() {
    cout << "Hello Lakisha Brooks!\n";
    cout << "\tHere are some of the C++ data types:\n";
    cout << "Size of char: " << sizeof(char) << " bytes\n";
    cout << "Size of bool: " << sizeof(bool) << " bytes\n";
    cout << "Size of short: " << sizeof(short) << " bytes\n";
    cout << "Size of int: " << sizeof(int) << " bytes\n";
    cout << "Size of long: " << sizeof(long) << " bytes\n";
    cout << "Size of long long: " << sizeof(long long) << " bytes\n";
    cout << "Size of float: " << sizeof(float) << " bytes\n";
    cout << "Size of double: " << sizeof(double) << " bytes\n";


    cout << "Thank you for reading about the size of different data types!" << endl;
    return 0;
}
