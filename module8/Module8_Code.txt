#include <iostream>
#include <fstream>
#include <string>

using namespace std;

// Step 1: Define the struct exactly as required
struct Patient {
    int id;
    string name;
    int age;
    string gender;
    string diagnosis;
};

// Helper function to swap two patients (used for sorting before binary search)
void swapPatients(Patient& a, Patient& b) {
    Patient temp = a;
    a = b;
    b = temp;
}

// Selection sort to keep the array sorted by Patient ID for Binary Search
void sortPatients(Patient* patients, int count) {
    for (int i = 0; i < count - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < count; j++) {
            if (patients[j].id < patients[minIdx].id) {
                minIdx = j;
            }
        }
        if (minIdx != i) {
            swapPatients(patients[i], patients[minIdx]);
        }
    }
}

int main() {
    // Step 2: Dynamic Array Requirement using pointers
    int maxCapacity = 100;
    Patient* patients = new Patient[maxCapacity];
    int count = 0; // Variable to track the total number of patients

    int choice = 0;

    // Step 3: Create Menu Loop
    while (choice != 6) {
        cout << "\nHospital Patient System" << endl;
        cout << "1. Add Patient" << endl;
        cout << "2. Display Patients" << endl;
        cout << "3. Search Patient" << endl;
        cout << "4. Update Patient" << endl;
        cout << "5. Save Patients" << endl;
        cout << "6. Exit" << endl;
        cout << "Enter choice: ";
        
        if (!(cin >> choice)) {
            cout << "Invalid menu input! Please enter a number." << endl;
            cin.clear();
            cin.ignore(1000, '\n');
            continue;
        }

        // Step 4: Add Patient
        if (choice == 1) {
            if (count >= maxCapacity) {
                cout << "Database full!" << endl;
                continue;
            }

            cout << "Enter Patient ID: ";
            while (!(cin >> patients[count].id)) {
                cout << "Invalid ID! Enter a number: ";
                cin.clear();
                cin.ignore(1000, '\n');
            }

            cin.ignore(1000, '\n'); // Clear buffer for string reads
            cout << "Enter Name: ";
            getline(cin, patients[count].name);

            cout << "Enter Age: ";
            while (!(cin >> patients[count].age)) {
                cout << "Invalid age! Enter a number: ";
                cin.clear();
                cin.ignore(1000, '\n');
            }

            cin.ignore(1000, '\n');
            cout << "Enter Gender: ";
            getline(cin, patients[count].gender);

            cout << "Enter Diagnosis: ";
            getline(cin, patients[count].diagnosis);

            count++;
            cout << "Patient added successfully!" << endl;
            
            // Keep the array sorted by ID immediately after insertion so binary search works
            sortPatients(patients, count);
        }
        // Step 5: Display All Patients
        else if (choice == 2) {
            if (count == 0) {
                cout << "No patient records available." << endl;
                continue;
            }
            cout << "\n--- Current Patient Records ---" << endl;
            for (int i = 0; i < count; i++) {
                cout << "ID: " << patients[i].id 
                     << " | Name: " << patients[i].name 
                     << " | Age: " << patients[i].age 
                     << " | Gender: " << patients[i].gender 
                     << " | Diagnosis: " << patients[i].diagnosis << endl;
            }
        }
        // Step 6: Search for patient using Binary Search
        else if (choice == 3) {
            if (count == 0) {
                cout << "Database is empty." << endl;
                continue;
            }

            int searchId;
            cout << "Enter patient ID to search: ";
            cin >> searchId;

            int low = 0;
            int high = count - 1;
            int foundIdx = -1;

            // Binary Search Algorithm
            while (low <= high) {
                int mid = low + (high - low) / 2;
                if (patients[mid].id == searchId) {
                    foundIdx = mid;
                    break;
                }
                else if (patients[mid].id < searchId) {
                    low = mid + 1;
                }
                else {
                    high = mid - 1;
                }
            }

            if (foundIdx != -1) {
                cout << "\nPatient Found!" << endl;
                cout << "Name: " << patients[foundIdx].name << endl;
                cout << "Age: " << patients[foundIdx].age << endl;
                cout << "Diagnosis: " << patients[foundIdx].diagnosis << endl;
            } else {
                cout << "Patient not found." << endl;
            }
        }
        // Step 7: Ability to update patient
        else if (choice == 4) {
            if (count == 0) {
                cout << "Database is empty." << endl;
                continue;
            }

            int updateId;
            cout << "Enter patient ID to update: ";
            cin >> updateId;

            int targetIdx = -1;
            for (int i = 0; i < count; i++) {
                if (patients[i].id == updateId) {
                    targetIdx = i;
                    break;
                }
            }

            if (targetIdx != -1) {
                cin.ignore(1000, '\n');
                cout << "Enter new diagnosis: ";
                getline(cin, patients[targetIdx].diagnosis);
                cout << "Diagnosis updated successfully!" << endl;
            } else {
                cout << "Patient ID not found." << endl;
            }
        }
        // Step 8: Store patient data to patients.txt
        else if (choice == 5) {
            ofstream outFile("patients.txt");
            if (!outFile) {
                cout << "Error creating file!" << endl;
                continue;
            }

            // Outputs formatted to this style: 1001 John 42 Male Flu
            for (int i = 0; i < count; i++) {
                outFile << patients[i].id << " " 
                        << patients[i].name << " " 
                        << patients[i].age << " " 
                        << patients[i].gender << " " 
                        << patients[i].diagnosis << endl;
            }
            outFile.close();
            cout << "Patient records safely stored to patients.txt!" << endl;
        }
        // Step 8 (Exit Program)
        else if (choice == 6) {
            cout << "Good Bye!" << endl;
        }
        else {
            cout << "Invalid option chosen." << endl;
        }
    }

    // Clean up allocated memory dynamically
    delete[] patients;
    return 0;
}
